# 插入排序

```js
Array.prototype.insertSort = function () {
	let arr = this.slice(0);

	for (var i = 1; i < arr.length; i++) {
		var j = i - 1;
		var num = arr[i];
		while (j >= 0 && arr[j] > num) {
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = num;
	}
	return arr;
};

var a = [
	3, 5, 3, 0, 8, 6, 1, 5, 8, 6, 2, 4, 9, 4, 7, 0, 1, 8, 9, 7, 3, 1, 2, 5, 9, 7,
	4, 0, 2, 6,
];

console.log(a.insertSort());
```
